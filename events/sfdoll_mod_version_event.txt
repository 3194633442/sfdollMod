country_event = {

	hide_window = yes
	is_triggered_only = yes

	}

		if = {
			limit = { has_ascension_perk = ap_sfdoll_reverse_collapse }
			add_research_option = tech_sfdoll_collapse_liquid_produce
		}
	}
}
